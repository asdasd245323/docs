---
title: com.gigs.invoice.voided
description: Triggered when an invoice is voided and can no longer be paid.
---

<ResponseField name="object" type="string" required>
  Type of object is always `event`.

  Allowed values: `event`
</ResponseField>

<ResponseField name="id" type="string" required>
  Unique identifier for the event.
</ResponseField>

<ResponseField name="actor" type="object" required>
  <Expandable title="properties">
    <ResponseField name="type" type="string" required>
      Allowed values: `user`, `apiKey`, `member`, `system`
    </ResponseField>
    <ResponseField name="user" type="string">
      Present if type is `user`.
    </ResponseField>
    <ResponseField name="apiKey" type="string">
      Present if type is `apiKey`.
    </ResponseField>
    <ResponseField name="member" type="string">
      Present if type is `member`.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="data" type="object" required>
  Invoices describe the amount due to a user for a given subscription period, including any relevant taxes and fees.
</ResponseField>

<ResponseField name="datacontenttype" type="string" required default="application/json">
  The RFC 2046 content-type.
</ResponseField>

<ResponseField name="project" type="string" required>
  Unique identifier for the project where the event occurred.
</ResponseField>

<ResponseField name="source" type="string" required default="https://api.gigs.com">
  URI identifying the event source.
</ResponseField>

<ResponseField name="specversion" type="string" required default="1.0">
  The CloudEvents spec version.
</ResponseField>

<ResponseField name="time" type="string" required>
  Time when the event occurred.
</ResponseField>

<ResponseField name="type" type="string" required>
  Type is always `com.gigs.invoice.voided`.

  Allowed values: `com.gigs.invoice.voided`
</ResponseField>

<ResponseField name="version" type="string" required default="2026-01-29">
  API version used to serialize the data and the event itself.
</ResponseField>
